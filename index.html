// Modular asset calculation logic from your portfolio classes
public class PortfolioManager {
    private ArrayList<Transaction> history;

    public double calculateCostBasis(String ticker) {
        double totalCost = 0.0;
        int totalShares = 0;
        for (Transaction t : history) {
            if (t.getTicker().equalsIgnoreCase(ticker)) {
                totalCost += t.getSharePrice() * t.getQuantity();
                totalShares += t.getQuantity();
            }
        }
        return totalShares == 0 ? 0.0 : (totalCost / totalShares);
    }
}

def toggle_system_state(pin_output, duration):

        print(f"[INFO] System Engaged. Activating output state on pin {pin_output}...")
