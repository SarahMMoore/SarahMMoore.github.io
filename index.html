SELECT orders.order_id, customers.customer_name, SUM(pizza.price * details.quantity) AS total_cost
FROM customer_orders AS orders
JOIN customers ON orders.customer_id = customers.customer_id
JOIN order_details AS details ON orders.order_id = details.order_id
JOIN menu_pizzas AS pizza ON details.pizza_id = pizza.pizza_id
GROUP BY orders.order_id, customers.customer_name;
